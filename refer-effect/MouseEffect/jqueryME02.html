const cursor = $('.cursor');
const follower = $('.cursor-follower');

//마우스 움직임
$(window).mousemove(function (e) {
    //gsap
    gsap.to(cursor, { duration: 0.3, left: e.pageX - 5, top: e.pageY - 5 });
    gsap.to(follower, { duration: 0.8, left: e.pageX - 15, top: e.pageY - 15 });
});

//hover 효과
$('.mouseCont em').hover(
    function () {
        cursor.addClass('active');
        follower.addClass('active');
    },
    function () {
        cursor.removeClass('active');
        follower.removeClass('active');
    }
);
$('.info').hover(
    function () {
        cursor.addClass('active2');
        follower.addClass('active2');
    },
    function () {
        cursor.removeClass('active2');
        follower.removeClass('active2');
    }
);

$(window).mousemove(function (e) {
    //페이지 기준
    $('.pageX').text(e.pageX + 'px');
    $('.pageY').text(e.pageY + 'px');
});
