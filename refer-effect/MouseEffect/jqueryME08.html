    color: orange;

.cursor {
    position: absolute;
    left: 0;
    top: 0;
    width: 10px;
    height: 10px;
    border-radius: 50%;
    z-index: 9999;
    background-color: rgba(255, 255, 255, 1);
    /* 아무것도 못하게 해줌 */
    user-select: none;
    pointer-events: none;
    transition: transform .3s;
}

.cursor.active {
    transform: scale(20);
    border: 1px solid orange;
    mix-blend-mode: difference;
}

const width = $('.cursor').width() / 2;
const height = $('.cursor').height() / 2;
let text = 'difference';

//마우스 움직임
$(window).mousemove(function (e) {
    gsap.to('.cursor', { duration: 0.4, left: e.pageX - width, top: e.pageY - height });
});

//hover 효과
$('.mouseCont em').hover(
    function () {
        $('.cursor').addClass('active');
        $('.cursor.active').css('mix-blend-mode', text);
    },
    function () {
        $('.cursor').removeClass('active');
        $('.cursor').css('mix-blend-mode', 'normal');
    }
);

//출력용
$('.info.left.list li').click(function () {
    $('.info.left.list li').removeClass('active');
    $(this).addClass('active');
    text = $(this).text();
});
